function enlist() { const username = document.getElementById('username').value; fetch('/api/enlist', { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify({ username }) }) .then(response => response.json()) .then(data => { // Handle success console.log('Success:', data); }) .catch((error) => { // Handle error console.error('Error:', error); }); }
